pipeline {
    agent any
    environment {
        ARTIFACT_BUCKET = 'devsecops-pipeline-artifacts-waliur'
        AWS_DEFAULT_REGION = 'us-east-1'
    }
    stages {
        stage('Local Workspace Preparation') {
            steps {
                echo 'Cleaning workspace and simulating source code checkout...'
                sh 'rm -f *.tf *.tar.gz'
            }
        }
        stage('Inject Infrastructure Code') {
            steps {
                echo 'Injecting active infrastructure design files into pipeline context...'
                // Native Jenkins utility that writes clean text without shell interpolation risks
                writeFile file: 'main.tf', text: '''resource "aws_s3_bucket" "vulnerable_bucket" {
  bucket = "vulnerable-test-bucket-sun-2026"
}'''
            }
        }
        stage('Infrastructure SAST Gate') {
            steps {
                echo 'Executing Checkov Static Application Security Testing against IaC manifests...'
                // The --soft-fail flag ensures checkov flags errors but lets us demonstrate the pipeline failure flow cleanly
                sh 'checkov -d . --framework terraform --compact'
            }
        }
        stage('Secure Archive Deployment') {
            steps {
                echo 'Vulnerability scanning stage evaluated. Archiving infrastructure artifacts...'
                sh 'tar -czf terraform-deployment.tar.gz main.tf'
                sh "aws s3 cp terraform-deployment.tar.gz s3://${env.ARTIFACT_BUCKET}/builds/terraform-deployment.tar.gz"
            }
        }
    }
}
